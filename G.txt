#include <stdio.h>
int count = 0;
void boxes(int arr[],int n,int b){
  if (b > 0)
  {
    /* code */
  
  for (int i = 0; i < n; ++i)
  {
    for (int j = i + 1; j < n; ++j)
    {
      if (arr[i] + arr[j] == b && (arr[i] != -1 || arr[j] != -1))
      {
        
        arr[j] = -1;
        arr[i] = -1;
        count++;
        break;
      }
    }
  }boxes(arr,n,b-1);
 }
}
int main(int argc, char const *argv[])
{
  int n,b;
  scanf("%d",&n);
  int arr[n];
  for (int i = 0; i < n; ++i)
  {
    scanf("%d ",&arr[i]);
  }
  scanf("%d",&b);
  boxes(arr,n,b);
  printf("%d",count);
  return 0;
}
